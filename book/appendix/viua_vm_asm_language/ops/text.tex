\begin{lstlisting}
text Rd %*\emph{<txt>}*)

\end{lstlisting}

\paragraph*{Opis} Wariant pierwszy konstruuje w rejestrze \texttt{Rd} tekst o wartości \emph{\texttt{txt}}.
